#include<stdio.h>
int main()
{
   int mark;
   printf("enter the mark:");
   scanf("%d",&mark);
   if(mark>=80)
   {
       printf("A");
   }
   else if
       (mark>=60)
   {
       printf("B");
   }
   
   else if(mark>=40)
   {
       printf("c");
   }
   else 
   {
       printf("fail");
   }
   
   
return 0;
}
